        //
            [DeleteAuthorize()]
        public ActionResult Delete(int id = 0)
        {
            if (id == 0)
            {
                return new HttpStatusCodeResult(HttpStatusCode.BadRequest);
            }

            {entityType} item = {entityType}Service.GetSingle(id);
            if (item == null)
            {
                return HttpNotFound();
            }


            return View(item);
        }

        [HttpPost, ActionName("Delete")]
        [ValidateAntiForgeryToken]
            [DeleteAuthorize()]
        public ActionResult DeleteConfirmed(int id)
        {

            {entityType} {entityType} = {entityType}Service.GetSingle(id);
            if ({entityType} == null)
            {
                return HttpNotFound();
            }
            try
            {
                {entityType}Service.Delete{entityType}ById(id);
                return RedirectToAction("Index");
            }
            catch (Exception ex)
            {
                Logger.Error(ex, "Unable to delete item:" + ex.StackTrace, {entityType});
                ModelState.AddModelError("", "Unable to save changes. Try again, and if the problem persists see your system administrator.");
            }

            return View({entityType});

        }
